LOOK FOR WRITABLE FILES
	find . -type f -perm -u=w
##########################################################################################
FIND BROKEN LINKS
	for f in $(find . -type l); do if [ ! -e "$f" ]; then echo "$f"; fi; done

	for i in `ls helloworld-demo.jar`; do echo $i; jar -tvf $i | grep -i HelloWorld.class; done

	for i in `ls *.jar`; do echo $i; jar -tvf $i | grep -i HelloWorld.class; done
